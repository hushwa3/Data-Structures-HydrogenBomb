#include <stdio.h>

int main() {
    int size;
    
    printf("Enter the size of the array: ");
    scanf("%d", &size);
    
    int arr[size];
    
    printf("Enter the elements of the array:\n");
    for (int i = 0; i < size; i++) {
        printf("Element %d: ", i);
        scanf("%d", &arr[i]);
    }
    
    int position;
    char choice = 'y';
    
    while (choice == 'y' || choice == 'Y') {
        printf("Enter the position to delete (0 - %d): ", size-1);
        scanf("%d", &position);
        
        if (position < 0 || position >= size) {
            printf("Invalid position!\n");
        } else {
            for (int i = position; i < size-1; i++) {
                arr[i] = arr[i+1];
            }
            size--;
            
            printf("Array after deleting the element:\n");
            for (int i = 0; i < size; i++) {
                printf("%d ", arr[i]);
            }
            printf("\n");
        }
        
        printf("Do you want to delete another element? (y/n): ");
        scanf(" %c", &choice);
        
    }
    
    return 0;
}
